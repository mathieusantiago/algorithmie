    var iCompare = 0;

    iValMax = tDico.length - 1;

    sRecherche = document.getElementById("iVal").value;
    //Pour iCoupteur = 0 a 20 iCompteur++
    for (iCompteur = 0; iCompteur < 20; iCompteur++) {

            //iValMax = iValMax - arondi(iDifference / 2)
