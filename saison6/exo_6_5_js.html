function Exo_6_5_js() {
    var i, k;
    var N = [6];
    
    N[0] = 1;
    for(k = 1; k <= 6; k++){
      N[k] = N[k-1]+2; 
    }
    for(i =0; i <= 6; i++){
      document.getElementById('execut').innerHTML += N[i];
