        result.innerHTML = "La date saisie est invalide"

        if (dAnnee % 4 === 0 && dAnnee % 100 > 0 || dAnnee % 400 === 0) {

                result.innerHTML = "La date saisie est valide"
            }else{
                result.innerHTML = "La date saisie est invalide"
            }
        }else{
            if (dJour >= 1 && dJour <= 28) {
                result.innerHTML = "La date saisie est valide"
            }else{
                result.innerHTML = "La date saisie est invalide"
            }
        }
        
    }else if (dMois == 4 || dMois == 6 || dMois == 9 || dMois == 11) {
        if (dJour >= 1 || dJour <= 30) {
            result.innerHTML = "La date saisie est valide"

            result.innerHTML = "La date saisie est invalide"

    }else{
        if (dJour >= 1 || dJour <= 31) {
            result.innerHTML = "La date saisie est valide"
        }else{
            result.innerHTML = "La date saisie est invalide"
        }
